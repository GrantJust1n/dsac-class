    function encrypt() {
      let text = document.getElementById("text").value;
      let shift = parseInt(document.getElementById("number").value);
      let encryptedText = shiftValue(text, shift);
      document.getElementById("encryptedMessage").innerText = "Encrypted Message: " + encryptedText;
    }

    function decrypt() {
      let decryptedText = document.getElementById("text").value; 
      document.getElementById("decryptedMessage").innerText = "Decrypted Message: " + decryptedText ;
    }

  
    function shiftValue(text, shift) {
      let result = '';
      for (let i = 0; i < text.length; i++) {
        let charCode = text.charCodeAt(i);

        if (charCode >= 65 && charCode <= 90) {
          result += String.fromCharCode(((charCode - 65 + shift) % 26) + 65);
        }
        else if (charCode >= 97 && charCode <= 122) {
          result += String.fromCharCode(((charCode - 97 + shift) % 26) + 97);
        }
     else{
          result += text.charAt(i);
        }
      }
      return result;
    }
